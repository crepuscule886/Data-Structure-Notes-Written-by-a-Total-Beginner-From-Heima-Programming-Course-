###### 结构体与指针

​	因为结构体有可能是巨大的，所以我们使用指针来传递结构体的地址

​	pp指向一个point(类型)结构体

> [!NOTE]
>
> struct point  *pp

  *pp是取结构体struct point的地址

为pp所指向的结构体中的变量赋值

下图的（*pp）是解引用，等价于p所以

(*pp).x = p.x = pp-> x

> [!NOTE]
>
> （*pp）.x =10;                  为了方便书写c语言提供了另外一种书写方式                   pp- > = 10;
>
> (*pp).y = 5;														       pp-> = 5;
>

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
// 1. 自定义结构体类型：描述一个坐标点
struct point
{
    int x;  // 成员1：x坐标
    int y;  // 成员2：y坐标
};
int main(int argc, char const* argv[])
{
    // 2. 定义结构体变量 p
    struct point p;
    // 直接给结构体变量的成员赋值（语法：变量名.成员名）
    p.x = 5;
    p.y = 10;
    // 3. 定义结构体指针 pp（专门指向 struct point 类型的指针）
    struct point* pp;
    // 让指针指向结构体变量 p 的地址
    pp = &p;
    // 4. 通过指针修改结构体成员（第一种写法：解引用+点运算符）
    (*pp).x = 10;  // *pp 等价于 p，所以等同于 p.x=10
    (*pp).y = 5;   // 等同于 p.y=5
    // 5. 打印结果：直接访问变量
    printf("x=%d,y=%d\n", p.x, p.y);
    // 6. 打印结果：通过指针访问（第二种写法：箭头运算符 ->）
    printf("x=%d,y=%d\n", pp->x, pp->y);
    return 0;
}
```

#### 动态内存分配

在c语言里面，动态分配内存的基本步骤：

1.使用malloc（memory allocate）函数分配内存(刚需头文件stdlib)

> [!TIP]
>
> 这个函数是用于帮助我们在堆内存当中，开辟这么一片地址用的

void*malloc（size_t）     【这个数据类型size_t,如果要用于输出的的话，它的通配符是%zu】

(sizeof函数的返回值类型，就是size_t)

> [!TIP]
>
> 所以这个数据类型，就是告诉电脑要开辟多大的内存空间
>
> 返回时，返回的类型是内存指针
>
> 这个void*表示的是一个通用数据类型的指针（存入什么类型，就会返回什么类型的指针“指针就是地址”）

如果成功，会返回从堆内存上分配的内存指针（point*）

如果失败，会返回空指针（）也就是NULL[因为不知道你开辟这片内存空间要说啥，所以运用空指针，进行一个强制转换]

2.使用分配的内存

3.使用free函数释放内存

##### 动态内存分配案例

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>
int main()
{
	int* p;
	p = (int*)malloc(sizeof(int));
	*p = 15;
	printf("%d\n", *p);
	free(p);
	return 0;
}
```

同时我们也可以优化一下，写下崩溃时，还有处理野指针

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<stdlib.h>

int main()
{
	int* p = (int*)malloc(sizeof(int));
	
	// 优化1：判断malloc是否申请成功（防止空指针崩溃）
	if (p == NULL) {
		printf("内存分配失败！\n");
		return 1; // 异常退出
	}

	*p = 15;
	printf("%d\n", *p);
	
	free(p);
	// 优化2：释放内存后，将指针置空，避免野指针
	p = NULL;

	return 0;
}
```

### 链表和顺序表

#### 	顺序表

用一组连续的内存单元依次存储线性表的各个元素，也就是说，逻辑上相邻的元素，实际的物理存储空间也是连续的（类似数组）

所以实现顺序表，可以用数组来实现。

正常的声明数组数组来存储顺序表（都是一些相对简单的一些内容）

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<string.h>
#define Maxsize 100
typedef int aaa;  // 给int起别名aaa
// 顺序表结构体
typedef struct  
{
    aaa data[Maxsizea];  // 数组定义语法：[ ] 而不是 { }
    aaa length;
}Seqlist;
//尾部添加
//length 是用来表示顺序表中有多少个数据，也可以当作数组下表来用
//传入一个顺序表的指针
aaa appendelement(Seqlist* L,aaa e)
{
    if (L->length>=Maxsize)
    {
        printf("shunxubiaoyiman\n");
        return 0;
    }
    L->data[L->length] = e;//即一开始data值是0.此时就是把e赋值到数组为0的下标那里
    L->length++;//加入后扩大顺序表的容量
    return 1;//1是成功，0是失败
}
//顺序表-遍历
void listElement(Seqlist* L)
{
    for (aaa i = 0; i < L->length; i++)
    {
        printf("%d", L->data[i]);
        printf("\n");
    }
}
// 顺序表初始化
void InitList(Seqlist* L)  // 函数名规范：Init 不是 int
{
    L->length = 0;  // 长度置0，表示空表
}
//插入顺序表
aaa selectElement(Seqlist* L, aaa x, aaa e)//这个x代表的位置就是你要插入的地方
{
    if (x <= L->length)/*判断：插入位置 是否合法
        比如现在有 5 个元素，只能插在 1~5 的位置*/
    {
        for (aaa i = L->length - 1; i >= x - 1; i--)
        {

            L->data[i + 1] = L->data[i];//把第i个位置的东西赋值给i+1个位置的
        }
```

```c
// 1. 编译器兼容配置（仅Windows VS编译器需要）
#define _CRT_SECURE_NO_WARNINGS

// 2. 导入C语言标准库

#define Maxsize 100

// 4. 定义顺序表存储的【元素类型】（通用化设计）
typedef int ElemType;

// 5. 定义顺序表结构体
typedef struct {
    ElemType* data;   // 指针：指向存储数据的动态数组
    int length;       // 变量：记录顺序表当前有多少个元素

// 6. 顺序表初始化函数：创建一个空的顺序表
Seqlist* initList() {
    // 第一步：给顺序表结构体本身申请内存
    Seqlist* L = (Seqlist*)malloc(sizeof(Seqlist));

    // 判断：结构体内存是否申请成功
    if (L == NULL) {
        printf("结构体内存分配失败！\n");
        return NULL;
    }

    // 第二步：给【存储数据的数组】申请内存（最大容量Maxsize）
    L->data = (ElemType*)malloc(sizeof(ElemType) * Maxsize);

    // 判断：数据数组内存是否申请成功
    if (L->data == NULL) {
        printf("数据数组内存分配失败！\n");
        free(L);  // 释放已申请的结构体内存，防止内存泄漏
        return NULL;
    }

    // 第三步：初始化顺序表为【空表】
    L->length = 0;

    // 返回创建好的顺序表指针
    return L;
}

// 7. 主函数：程序入口，测试顺序表

    // 调用初始化函数，创建一个空顺序表
    Seqlist* L = initList();

        printf("顺序表初始化成功！当前长度：%d\n", L->length);
    }

    // 8. 释放动态内存（必须写！防止内存泄漏）
    free(L->data);  // 先释放数据数组
```

```c
Node* findCycleEntry(Node* head)
{
    Node *fast = head;
    Node *slow = head;

    // 1. 快慢指针判断是否有环
    while (fast != NULL && fast->next != NULL)
    {
        fast = fast->next->next;  // 快指针：步长 2

            {
                cycleLen++;
                p = p->next;
            }

            fast = head;
            slow = head;
            while (fast != slow)
            {
                fast = fast->next;
                slow = slow->next;
            }
```

#### 1. **后序遍历（左 → 右 → 根）**（图中讲解的遍历方式）

**结果：`K H D E B I F J G C A`**

- 递归逻辑：先遍历左子树，再遍历右子树，最后访问当前根节点
- 验证：最底层的叶子节点 K 最先输出，整棵树的根节点 A 最后输出

#### 2. **前序遍历（根 → 左 → 右）**

**结果：`A B D H K E C F I G J`**

- 递归逻辑：先访问当前根节点，再遍历左子树，最后遍历右子树
- 验证：整棵树的根节点 A 最先输出

#### 3. **中序遍历（左 → 根 → 右）**

**结果：`H K D B E A I F C G J`**

- 递归逻辑：先遍历左子树，再访问当前根节点，最后遍历右子树
- 验证：对于二叉搜索树，中序遍历结果是有序序列

哈夫曼树的定义

带权路径长度wpl最小的二叉树称为哈夫曼树

构造哈夫曼树

1.先把有权值的叶子节点按照从小到大的顺序排列成一个有序序列

2.取两个最小权值的结点作为一个新结点N1的子结点

3.将结点N1替换刚刚取出的两个结点，并加入到有序序列中

4.重复步骤2，将n1和下一个结点作为一个新结点N2的子结点

5.将结点N2替换刚刚取出的两个结点。并加入有序序列中

后面基本都是重复

哈夫曼编码就是哈夫曼树左分叉 为0右编码为1

它的核心特点是在编码方案中，没有任何一个编码是其他编码的前缀，即没有任何编码与另一个编码的开始部分相同。这种属性使得前缀编码在数据压缩和传输中非常有用，因为它可以确保编码的唯一性和解码的准确性
